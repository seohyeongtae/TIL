
	/* TABLE 초기화  */
DROP SEQUENCE BULLETIN_SEQ;
DROP SEQUENCE SALES_SEQ;
DROP SEQUENCE SALES_SEQ;

DROP TABLE BULLETIN;
DROP TABLE SALES;
DROP TABLE RECEIPT;
DROP TABLE TAB;
DROP TABLE MENU;
COMMIT;







	/* TAB TABLE 구성*/
CREATE TABLE TAB(
ID VARCHAR2(10) PRIMARY KEY,
PWD VARCHAR2(10),
NAME NUMBER(3),
ADMIN_ID VARCHAR2(10),
NUM NUMBER(10)
);
COMMIT;



	/* MENU TABLE 구성*/
CREATE TABLE MENU(
ID VARCHAR2(10) PRIMARY KEY,
NAME VARCHAR2(20),
PRICE NUMBER(10),
TSALES NUMBER(10),
CATEGORY NUMBER(10),
IMG1 VARCHAR2(20),
IMG2 VARCHAR2(20),
IMG3 VARCHAR2(20)
);
COMMIT;



	/* RECEIPT TABLE 구성 */
CREATE TABLE RECEIPT(
ID VARCHAR2(10) PRIMARY KEY,
TAB_ID VARCHAR2(10),
STRING VARCHAR2(20),
PAYMENT NUMBER(2),
TOTAL NUMBER(10)
);
COMMIT;

	/* SALES TABLE 구성*/
CREATE TABLE SALES(
ID VARCHAR2(10) PRIMARY KEY,
MENU_ID VARCHAR2(20),
RECEIPT_ID VARCHAR2(10),
QT NUMBER(10),
S_SERVICE NUMBER(2),
S_PRICE NUMBER(10)
);
COMMIT;

COMMIT;

	/* BULLETIN TABLE 구성*/
CREATE TABLE BULLETIN(
ID NUMBER(5) PRIMARY KEY,
TAB_ID VARCHAR2(10),
TITLE VARCHAR2(20),
CONTENT VARCHAR2(100),
AUTHOR VARCHAR2(10),
REGDATE DATE,
COMMENT_ID NUMBER(5),
IMG VARCHAR2(20)
);


/*
ALTER TABLE BULLETIN ADD CONSTRAINT

COMMIT;
*/
